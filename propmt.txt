public IActionResult PatientsGrid()
{
    var patients = _patientService.GetAll().Select(p => new PatientDto {
        Id = p.Id,
        FullName = p.FullName,
        Phone = p.Phone,
        Age = p.Age,
        TotalDue = p.TreatmentPlans.Sum(tp => tp.Items.Sum(i => i.LineTotal))
    }).ToList();

    return PartialView("_PatientsGrid", patients);
}
